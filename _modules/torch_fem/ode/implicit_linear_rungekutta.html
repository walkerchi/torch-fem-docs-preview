[docs]    def step(self, t0, u0, dt):
        """

        .. math::

        Parameters
        ----------
        t0 : float
            initial time
        u0 : torch.Tensor
            initial value of shape :math:`[D]` where D is the dimension of the problem
        dt : float
            time step
        """
        assert u0.dim() == 1, f"expected u0 to be 1D tensor, got {u0.dim()}"
        D = u0.shape[0]
        h = dt 
        ts = t0 + dt * self.c
        lhs = [[None for _ in range(self.s)] for _ in range(self.s)]  
        rhs = [None for _ in range(self.s)]
        use_sparse = None
        for i in range(self.s):
            Ai = self.forward_A(ts[i])
            Bi = self.forward_B(ts[i])
            Mi = self.forward_M(ts[i])
            if i == 0:
                use_sparse = isinstance(Mi, SparseMatrix) or isinstance(Ai, SparseMatrix)
            else:
                if use_sparse:
                    assert not isinstance(Ai, torch.Tensor), f"expected A to be SparseMatrix or None, got {type(Ai)}"
                    assert not isinstance(Mi, torch.Tensor), f"expected M to be SparseMatrix or None, got {type(Mi)}"
                else:
                    assert not isinstance(Ai, SparseMatrix), f"expected A to be torch.Tensor or None, got {type(Ai)}"
                    assert not isinstance(Mi, SparseMatrix), f"expected M to be torch.Tensor or None, got {type(Mi)}"
            
            if Mi is None:
                if use_sparse:
                    Mi = SparseMatrix.eye(D)
                else:
                    Mi = torch.eye(D)
            elif isinstance(Mi, (int, float)):
                if use_sparse:
                    Mi = SparseMatrix.eye(D, value=Mi)
                else:
                    Mi = torch.eye(D) * Mi

            for j  in range(self.s):
                if Ai is None:
                    lhs[i][j] = - h  * self.a[i, j]  
                else:
                    lhs[i][j] = - h  * self.a[i, j] * Ai 
                if i == j and Mi is not None:
                    if Ai is None or isinstance(Ai, (int, float)): # lhs[i][j] of shape torch.Size([])
                        if use_sparse:
                            lhs[i][j] = SparseMatrix.eye(D, value=lhs[i][j]) + Mi
                        else:
                            lhs[i][j] = torch.eye(D) * lhs[i][j] + Mi
                    else:
                        lhs[i][j] = Mi + lhs[i][j] 

            if Ai is None and Bi is None:
                rhs[i] = u0
            elif Ai is None:
                rhs[i] = Bi
            else:
                if isinstance(Ai, (int, float)):
                    if use_sparse:
                        Ai = SparseMatrix.eye(D, value=Ai)
                    else:
                        Ai = torch.eye(D) * Ai
                if Bi is None:
                    rhs[i] = Ai @ u0 
                else:
                    rhs[i] = Bi + Ai @ u0 

        # combine lhs and rhs
        if use_sparse:
            lhs = SparseMatrix.combine(lhs)
        else:
            lhs = torch.cat([
                torch.cat(lhs[i], 1) for i in range(self.s)
            ], 0)
        rhs = torch.cat(rhs, 0)

        # solve linear system
        if use_sparse:
            k = lhs.solve(rhs)
        else:
            k = torch.linalg.solve(lhs, rhs)

        k = k.reshape(self.s, D)
        u = u0 + h * self.b @ k
    
        return u
